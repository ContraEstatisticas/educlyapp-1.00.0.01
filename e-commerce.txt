  noRes.style.display='none';
  grid.innerHTML=list.map(p=>`
    <div class="product-card" onclick="openProduct(${p.id})">
      <div class="product-img">
        <div class="product-img-fill" style="background:${p.gradient}"></div>
        ${p.badge?`<div class="badge badge-${p.badge}">${p.badge==='novo'?'Novo':'Sale'}</div>`:''}
        <div class="product-actions">
          <button class="pa-btn" onclick="event.stopPropagation();quickAdd(${p.id})">Adicionar</button>
          <button class="pa-btn dark" onclick="event.stopPropagation();openProduct(${p.id})">Ver mais</button>
        </div>
      </div>
      <div class="product-info">
        <div class="product-name">${p.name}</div>
        <div class="product-sub">${p.sub}</div>
        <div>
          <span class="product-price">R$ ${p.price.toLocaleString('pt-BR',{minimumFractionDigits:0})}</span>
          ${p.oldPrice?`<span class="product-price-old">R$ ${p.oldPrice.toLocaleString('pt-BR',{minimumFractionDigits:0})}</span>`:''}
        </div>
      </div>
    </div>
  `).join('');
}

function openProduct(id){
  const p=products.find(x=>x.id===id);
  if(!p)return;
  currentProduct=p; selectedSize=null; selectedColor=null;
  document.getElementById('modalImgFill').style.background=p.gradient;
  document.getElementById('modalTag').textContent=p.sub;
  document.getElementById('modalName').textContent=p.name;
  document.getElementById('modalPrice').innerHTML='R$ '+p.price.toLocaleString('pt-BR',{minimumFractionDigits:0})+(p.oldPrice?`<span>R$ ${p.oldPrice.toLocaleString('pt-BR',{minimumFractionDigits:0})}</span>`:'');
  document.getElementById('modalDesc').textContent=p.desc;
  document.getElementById('modalSizes').innerHTML=p.sizes.map(s=>`<button class="size-btn" onclick="selectSize('${s}',this)">${s}</button>`).join('');
  document.getElementById('modalColors').innerHTML=p.colors.map(c=>`<div class="color-dot" style="background:${c}" onclick="selectColor('${c}',this)"></div>`).join('');
  document.getElementById('addCartBtn').disabled=false;
  document.getElementById('modalOverlay').classList.add('open');
  document.body.style.overflow='hidden';
}

function addToCart(){
  if(!currentProduct)return;
  if(!selectedSize){showToast('Selecione um tamanho');return;}
  const existing=cart.find(i=>i.id===currentProduct.id&&i.size===selectedSize&&i.color===selectedColor);
  if(existing){existing.qty++;}
  else{cart.push({...currentProduct,size:selectedSize,color:selectedColor||currentProduct.colors[0],qty:1});}
  updateCartBadge();
  closeModal();
  showToast(currentProduct.name+' adicionado ao carrinho');
  renderCart();
}

function quickAdd(id){
  const p=products.find(x=>x.id===id);
  if(!p)return;
  const existing=cart.find(i=>i.id===p.id);
  if(existing){existing.qty++;}
  else{cart.push({...p,size:p.sizes[0],color:p.colors[0],qty:1});}
  updateCartBadge();
  showToast(p.name+' adicionado');
  renderCart();
}

function updateCartBadge(){
  const total=cart.reduce((s,i)=>s+i.qty,0);
  const badge=document.getElementById('cartBadge');
  badge.textContent=total;
  badge.classList.toggle('show',total>0);
}

function openCart(){
  renderCart();
  document.getElementById('cartDrawer').classList.add('open');
  document.getElementById('cartOverlay').classList.add('open');
  document.body.style.overflow='hidden';
}

function closeCart(){
  document.getElementById('cartDrawer').classList.remove('open');
  document.getElementById('cartOverlay').classList.remove('open');
  document.body.style.overflow='';
}

function removeFromCart(idx){
  cart.splice(idx,1);
  updateCartBadge();
  renderCart();
}

function changeQty(idx,delta){
  cart[idx].qty+=delta;
  if(cart[idx].qty<=0){cart.splice(idx,1);}
  updateCartBadge();
  renderCart();
}

function renderCart(){
  const el=document.getElementById('cartItems');
  const footer=document.getElementById('cartFooter');
  if(!cart.length){
    el.innerHTML='<div class="cart-empty">Seu carrinho está vazio.<br>Explore a coleção e adicione peças.</div>';
    footer.style.display='none';
    return;
  }
  footer.style.display='flex';
  el.innerHTML=cart.map((item,i)=>`
    <div class="cart-item">
      <div class="cart-item-img"><div class="cart-item-img-fill" style="background:${item.gradient}"></div></div>
      <div class="cart-item-info">
        <div class="cart-item-name">${item.name}</div>
        <div class="cart-item-meta">${item.size} &middot; <span style="display:inline-block;width:10px;height:10px;border-radius:50%;background:${item.color};border:1px solid rgba(0,0,0,0.1);vertical-align:middle"></span></div>
        <div class="cart-qty">
          <button class="qty-btn" onclick="changeQty(${i},-1)">−</button>
          <span class="qty-num">${item.qty}</span>
          <button class="qty-btn" onclick="changeQty(${i},1)">+</button>
        </div>
        <div class="cart-item-price">R$ ${(item.price*item.qty).toLocaleString('pt-BR',{minimumFractionDigits:0})}</div>
      </div>
      <button class="cart-item-remove" onclick="removeFromCart(${i})">×</button>
    </div>
  `).join('');
  const total=cart.reduce((s,i)=>s+i.price*i.qty,0);
  document.getElementById('cartTotal').textContent='R$ '+total.toLocaleString('pt-BR',{minimumFractionDigits:2});
}

function showToast(msg){
  const t=document.getElementById('toast');
  t.textContent=msg;
  t.classList.add('show');
  clearTimeout(t._timer);
  t._timer=setTimeout(()=>t.classList.remove('show'),2800);
}
